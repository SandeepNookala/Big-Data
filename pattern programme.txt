
#pattern programme - rectangle shape:
#===================================

for i in range(5):
    for j in range(5):
        print('*',end='')
    print()
	


#pattern programme - Triangle:
#=============================
for i in range(5):
    for j in range(i+1):
        print('*',end='')
    print()

#pattern programme - inverted tringle

for i in range(5):
    for j in range(5-i):
        print('*',end='')
    print()
